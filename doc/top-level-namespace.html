post('/searching') do   
    searched_tag = params["Search"]
    if searched_tag.length != 0
        if check_tag(searched_tag) == true
            $tagged_memes = search(searched_tag)
            redirect('/vault_search')
        else 
            flash[:notice] = "No such tag!"
            redirect('/vault')
        end 
    else 
        flash[:warning] = "PLEASE ENTER A TAG!"
        redirect('/vault')
    end 
end
